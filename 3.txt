create table Dept (
    deptId int not null,
    deptName varchar(10),
    primary key (deptId)
);

create table Emp (
    empId int not null,
    empName varchar(10) not null,
    empSal int,
    empDeptId int,
    PRIMARY KEY (empId),
    FOREIGN KEY (empDeptId) REFERENCES Dept(deptId)
);
